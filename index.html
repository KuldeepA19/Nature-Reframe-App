// New Navigation Logic function showScreen(screenId) { // Hide all screens document.querySelectorAll('.screen').forEach(s => s.classList.remove('active-screen')); // Show selected screen document.getElementById(screenId).classList.add('active-screen'); } // Improved Audio Logic (Prevents overlapping sounds) function stopAllAudio() { const allAudios = document.querySelectorAll('audio'); allAudios.forEach(audio => { audio.pause(); audio.currentTime = 0; }); } function playTrack(type) { stopAllAudio(); const track = document.getElementById(`${type}Track`); if(track) track.play(); }
